    // -----------------------------
    // Generic reusable initialiser
    // -----------------------------
    function initRadioGroup(groupName) {

        const radios = Array.from(document.querySelectorAll(`input[name="${groupName}"]`));

        radios.forEach(radio => {

            // Click / selection
            radio.addEventListener("change", () => {
                updateRadioGroup(groupName, radio);
            });

            // Arrow-key navigation
            radio.addEventListener("keydown", e => {
                const index = radios.indexOf(radio);

                if (e.key === "ArrowRight" || e.key === "ArrowDown") {
                    e.preventDefault();
                    const next = radios[(index + 1) % radios.length];
                    next.checked = true;
                    updateRadioGroup(groupName, next);
                    next.focus();
                }

                if (e.key === "ArrowLeft" || e.key === "ArrowUp") {
                    e.preventDefault();
                    const prev = radios[(index - 1 + radios.length) % radios.length];
                    prev.checked = true;
                    updateRadioGroup(groupName, prev);
                    prev.focus();
                }
            });

        });
    }

    // -----------------------------
    // Generic state updater
    // -----------------------------
    function updateRadioGroup(groupName, selected) {

        const radios = document.querySelectorAll(`input[name="${groupName}"]`);

        radios.forEach(r => {
            const isSelected = r === selected;
            r.removeAttribute('checked');
            r.checked = isSelected;
            r.setAttribute("aria-checked", isSelected ? "true" : "false");
            r.setAttribute("tabindex", isSelected ? "0" : "-1");
        });
    }
</script>
