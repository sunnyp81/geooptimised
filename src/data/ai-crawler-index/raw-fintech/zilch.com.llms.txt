const ids = ["days", "hours", "minutes", "seconds"];
    const countdownTime = new Date("2025-06-27T00:00:00+01:00").getTime();

    const updateCountdown = () => {
      const now = Date.now();
      const diff = countdownTime - now;

      if (diff <= 0) {
        clearInterval(timer);
        ids.forEach(id => document.getElementById(id).textContent = "00");
        return;
      }

      const t = {
        days: Math.floor(diff / (1000 * 60 * 60 * 24)),
        hours: Math.floor((diff / (1000 * 60 * 60)) % 24),
        minutes: Math.floor((diff / (1000 * 60)) % 60),
        seconds: Math.floor((diff / 1000) % 60),
      };

      ids.forEach(id => {
        document.getElementById(id).textContent = String(t[id]).padStart(2, '0');
      });
    };

    updateCountdown(); // initial call to avoid 1s delay
    const timer = setInterval(updateCountdown, 1000);
    </script>

function dotSmoothSlide (parentElem, scrollElem, sliderElem, fullScreen){
    if(window.innerWidth <= 500 || fullScreen == "full"){
    parentElem.style.display = "flex"
    let len = sliderElem.children.length
            for(let i = 0; i < len; i++){
                const dot = document.createElement("div")
                dot.setAttribute("class", "dot")
                    // if(i == 0) { dot.classList.add("dot-active") }
                parentElem.appendChild(dot)
            }

    
    scrollElem.addEventListener("scroll", (e) => {
        let math = scrollElem.scrollLeft / window.innerWidth
        const dotElem = parentElem.children
        const alg = Math.round(math / 0.5) * 0.5;
        // console.log(alg)
        for(let i = 0; i < dotElem.length; i++){
                if(parseInt(alg) === i){
                    gsap.to(dotElem[parseInt(alg)], .1, {maxWidth:25})
                    
                }else{
                    gsap.to(dotElem[i], .1, {maxWidth:6})
                }
                
        }
        

    })
    }else if(fullScreen == "custom"){
        parentElem.style.display = "flex"
         const images = sliderElem.querySelectorAll('img');
        let len = images.length

            for(let i = 0; i < len; i++){
                const dot = document.createElement("div")
                dot.setAttribute("class", "dot")

                parentElem.appendChild(dot)
