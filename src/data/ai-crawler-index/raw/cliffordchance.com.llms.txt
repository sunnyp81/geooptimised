                //GET COMPONENT NAME FROM DOM
                function getComponentFromClick(el) {
                    if (!el) return '';

                    var a = (el.tagName === 'A' || el.tagName === 'BUTTON') ? el : (el.closest ? el.closest('a,button') : null);
                    if (!a) return '';

                //GET LINK NAME FROM DOM 
                function getDomLinkName(a) {
                    if (!a) return '';

                    //aria-label (best for icons)
                    var aria = (a.getAttribute('aria-label') || '').trim();
                    if (aria) return aria;

                    //Extract PRIMARY visible CTA text (ignore nested spans)
                    var primaryText = '';

                    for (var i = 0; i < a.childNodes.length; i++) {
                        var node = a.childNodes[i];

                        // Direct text node inside <a>
                        if (node.nodeType === Node.TEXT_NODE) {
                            var text = node.textContent.replace(/\s+/g, ' ').trim();
                            if (text) {
                                primaryText = text;
                                break;
                            }
                        }

                        // If first-level span, extract only its direct text (not nested span)
                        if (node.nodeType === Node.ELEMENT_NODE && node.tagName === 'SPAN') {
                            for (var j = 0; j < node.childNodes.length; j++) {
                                var child = node.childNodes[j];

                                if (child.nodeType === Node.TEXT_NODE) {
                                    var spanText = child.textContent.replace(/\s+/g, ' ').trim();
                                    if (spanText) {
                                        primaryText = spanText;
                                        break;
                                    }
                                }
                            }
                            if (primaryText) break;
                        }
                    }

                    if (primaryText) return primaryText;

                    //Fallback to title ONLY if no visible text found
                    var title = (a.getAttribute('title') || '').trim();
                    if (title) return title;

                    return '';
                }

                //DETERMINE LINK TYPE
                function inferLinkType(href) {
                    if (!href) return 'Internal';
                    var h = href.trim().toLowerCase();

                    // Anchor links (#section)
                    if (h.startsWith('#')) {
                        return 'Internal';
                    }
                    // Email links should be treated as Internal
                    if (h.startsWith('mailto:')) {
                        return 'Internal';
                    }

                    // Relative internal URLs
                    if (h.startsWith('/') || h.startsWith('./') || h.startsWith('../')) {
                        return 'Internal';
                    }

                    try {
                        // Absolute URLs
                        var url = new URL(h, window.location.origin);
                        var hostname = url.hostname.toLowerCase();

                        // Internal domain check
                        if (hostname.endsWith('cliffordchance.com')) {
                            return 'Internal';
                        }

                        return 'External';

                    } catch (e) {

                        // Fallback safety
                        return 'External';
                    }
                }

                //NORMALIZE DESTINATION URL
                function normalizeDestination(url, type) {
                    if (!url) return '';
                    var u = url.trim();

                    // Handle anchor links (#section)
                    if (u.startsWith('#')) {
                        return u;
                    }
                    // Preserve email links exactly as they are
                    if (u.toLowerCase().startsWith('mailto:')) {
                        return u;
                    }
                    u = u.replace(/^https?:\/\//i, '');

                    if (type === 'Internal') {
                        u = u.replace(/^[^\/]*cliffordchance\.com/i, '');
                        u = u.replace(/^\/?content\/cliffordchance/i, '');
                        if (!u.startsWith('/')) u = '/' + u;
                    }

                    u = u.replace(/\.html.*$/i, '');
                    // keep anchor destination if it is only hash
                    // Remove query parameters but keep anchors
                    u = u.split('?')[0];

                    return u;
                }

                function buildFinalValue(el, href) {

                    return [component, linkName, dest, type].join('|');
                }

            function toLowerTrim(v) {
                return (v == null ? '' : String(v)).trim().toLowerCase();
            }
            
            function isEmpty(v) {
                return !v || !String(v).trim();
            }
            
            /**
            * Returns selected option TEXT for a <select>. Example: "Banks", "London", "ALL"
            * If select doesn't exist or selection missing -> "".
            */
            function getSelectedText(selectId) {
                var el = document.getElementById(selectId);
                if (!el || !el.options || el.selectedIndex < 0) return '';
                return (el.options[el.selectedIndex].text || '').trim();
            }
            
            /**
            * Returns selected option VALUE for a <select>. Example: "sectors:banks", "/content/.../london"
            * If select doesn't exist -> "".
            */
            function getSelectedValue(selectId) {
                var el = document.getElementById(selectId);
                return el ? (el.value || '').trim() : '';
            }
            
            /**
            * Converts "ALL"/"" to "all", otherwise returns lowercase of label.
            */
            function normalizeDropdownText(selectedText) {
                var v = toLowerTrim(selectedText);
                return (v === '' || v === 'all' || v === 'all ') ? 'all' : v;
            }
            
            /**
            * Fallback normalization when only value is available (URL or select value)
            * - "sectors:banks" -> "banks"
            * - "legal_area:antitrust..." -> "antitrust..."
            */
            function normalizeTagValueFromValue(rawValue) {
                var v = toLowerTrim(rawValue);
                if (!v || v === 'all') return 'all';
                return v.indexOf(':') > -1 ? v.split(':').pop() : v;
            }
            
            /**
            * Office value might be a path -> last segment
            * - "/content/.../london" -> "london"
            */
            function normalizeOfficeFromValue(rawValue) {
                var v = toLowerTrim(rawValue);
                if (!v || v === 'all') return 'all';
                if (v.indexOf('/') > -1) v = v.split('/').pop();
                return v || 'all';
            }

            function getSearchStatus() {
                var path = window.location.pathname || '';
                var isPeoplePage = path.indexOf('/people_and_places/people.html') > -1 ||
                    path.indexOf('/people_and_places.html') > -1;
                // ---- People & Places status ----
                if (isPeoplePage) {
                    var $results = $('#section_results');
                    
                    // If section_results isn't present, treat as unsuccessful
                    if ($results.length === 0) return 'unsuccessful';
                    
                    // People results
                    var peopleCount = $results.find('article.article_result').length;
                    
                    // Offices results (your markup shows both patterns)
                    var officeCount =
                        $results.find('article.article_result_office').length +
                            $results.find('section.section_result_office').length;
                    
                    return (peopleCount + officeCount) > 0 ? 'successful' : 'unsuccessful';
                }
                
                // ---- Global search status ----
                return ($('ul.ul_searchlist li').length > 0) ? 'successful' : 'unsuccessful';
            }
            

            function searchClickPrep() {
                
                // ---------------- Defaults ----------------
                var sSearchTerm = '';
                var sSearchCategory = '';
                var sSearchStatus = 'unsuccessful';
                
                var sSearchSector = 'all';
                var sSearchPracticeArea = 'all';
                var sSearchOffice = 'all';
                
                // Extended search params
                var sFname = '';
                var sLname = '';
                var sOffice = '';
                
                
                    
                    
                    
                    
                    
                    
                    
                    
                    
                    // ---------------- Helpers ----------------
                    function clean(v) {
                        if (v === null || v === undefined) return '';
                        return String(v).trim();
                    }
                function lowerOr(v, fallback) {
                    v = clean(v).toLowerCase();
                    return v ? v : fallback;
                }
                function hasValue(v) {
                    return clean(v).length > 0;
                }
                
                var path = window.location.pathname || '';
                var isPeoplePage = path.indexOf('/people_and_places/people.html') > -1 ||
                    path.indexOf('/people_and_places.html') > -1;
                
                // Detect extended form search (fname/lname/tags/office pattern)
                var qs = window.location.search || '';
                var hasTags = (qs.indexOf('tags=') > -1) || (qs.indexOf('+tags=') > -1);
                var isExtendedPeopleSearch = isPeoplePage && (hasValue(sFname) || hasValue(sLname) || hasValue(sOffice) || hasTags) && !hasValue(sSearchTerm);
                
                var isPeopleMenuSearch = isPeoplePage && hasValue(sSearchTerm); // hover search lands with query=...
                var isGlobalSearch = !isPeoplePage; // global site search
                
                // ---------------- Term rules ----------------
                if (isExtendedPeopleSearch) {
                    // keyword = fname + lname, but no "unspecified" parts
                    var parts = [];
                    if (hasValue(sFname)) parts.push(clean(sFname));
                    if (hasValue(sLname)) parts.push(clean(sLname));
                    
                    if (parts.length === 0) {
                        sSearchTerm = 'Unspecified';
                    } else {
                        sSearchTerm = parts.join(' ').toLowerCase();
                    }
                } else {
                    // global + hover searches must block empty keyword
                    sSearchTerm = clean(sSearchTerm);
                    if (!sSearchTerm) {
                        return; // block empty searches -> fixes "None"
                    }
                    // (Requirement only demands lowercase for extended, but doing it here is safe/consistent)
                    sSearchTerm = sSearchTerm.toLowerCase();
                }
                
                // ---------------- Category rules ----------------
                if (isExtendedPeopleSearch || isPeopleMenuSearch) {
                    sSearchCategory = 'people'; // always
                } else {
                    // global search category normalize (keep your existing logic)
                    var categoryMap = {
                        "all": "All",
                        "people": "People",
                        "expertise": "Expertise",
                        "briefings": "Publications",
                        "alerters": "Alerters",
                        "news": "News",
                        "offices": "Offices",
                        "thought_leadership": "Thought Leadership"
                    };
                    
                    if (!sSearchCategory) {
                        sSearchCategory = 'All';
                    } else {
                        var rawCategory = decodeURIComponent(sSearchCategory);
                        
                        if (rawCategory.indexOf('/') > -1) {
                            rawCategory = rawCategory.split('/').pop();
                        }
                        
                        var key = rawCategory.toLowerCase();
                        if (categoryMap[key]) {
                            sSearchCategory = categoryMap[key];
                        } else {
                            sSearchCategory = rawCategory
                            .replace(/_/g, ' ')
                            .replace(/\b\w/g, function (l) { return l.toUpperCase(); });
                        }
                    }
                    
                    // match requirement formatting style -> lower
                    sSearchCategory = sSearchCategory.toLowerCase();
                    if (!sSearchCategory) sSearchCategory = 'all';
                }
                
                // ---------------- Sector/Practice/Office rules ----------------
                if (isExtendedPeopleSearch) {
                    
                    // Preferred: use selected TEXT (human-friendly reporting)
                    var sectorText = getSelectedText('industry');     // sector dropdown
                    var practiceText = getSelectedText('legalArea');  // practice dropdown
                    var officeText = getSelectedText('location');     // office dropdown

                    sSearchSector = normalizeDropdownText(sectorText);
                    sSearchPracticeArea = normalizeDropdownText(practiceText);
                    sSearchOffice = normalizeDropdownText(officeText);
                    
                    // Fallback: if selects not ready (loaded later), use VALUES instead
                    // This keeps tracking working even if JS runs before dropdowns render.
                    if (sSearchSector === 'all' && sSearchPracticeArea === 'all' && sSearchOffice === 'all') {
                        
                        var sectorVal = getSelectedValue('industry');
                        var practiceVal = getSelectedValue('legalArea');
                        var officeVal = getSelectedValue('location');
                        
                        sSearchSector = normalizeTagValueFromValue(sectorVal);
                        sSearchPracticeArea = normalizeTagValueFromValue(practiceVal);
                        sSearchOffice = normalizeOfficeFromValue(officeVal);
                        
                        // If still empty (extreme edge case), keep "all"
                        if (isEmpty(sSearchSector)) sSearchSector = 'all';
                        if (isEmpty(sSearchPracticeArea)) sSearchPracticeArea = 'all';
                        if (isEmpty(sSearchOffice)) sSearchOffice = 'all';
                    }
                    
                } else {
                    // Hover + global searches always default to all
                    sSearchSector = 'all';
                    sSearchPracticeArea = 'all';
                    sSearchOffice = 'all';
                }
                
                
                sSearchSector = lowerOr(sSearchSector, 'all');
                sSearchPracticeArea = lowerOr(sSearchPracticeArea, 'all');
                sSearchOffice = lowerOr(sSearchOffice, 'all');

                // ---------------- Status rules (never unspecified) ----------------
                sSearchStatus = getSearchStatus();

                // ---------------- Call updated searchClick ----------------
                searchClick(
                    sSearchTerm,
                    sSearchCategory,
                    sSearchSector,
                    sSearchPracticeArea,
                    sSearchOffice,
                    sSearchStatus
                );
            }
